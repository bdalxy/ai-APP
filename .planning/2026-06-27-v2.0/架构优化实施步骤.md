### 1.2 模块耦合分析

#### 问题1：chat_bridge/_core.py 是"上帝模块"

`_core.py` 的 `chat()` 和 `chat_stream_start()` 函数直接依赖了所有模块：

```python
# _core.py 中的 chat() 函数，直接耦合了所有子系统
def chat(user_input: str):
    # 直接调用插件管理器
    processed = _get_plugin_manager().pre_process(user_input)
    # 直接调用世界书注入
    _inject_context(player, user_input, orchestrator)  # 内部调用 _world_book._match_and_inject_for_all
    # 直接调用插件后处理
    reply = _get_plugin_manager().post_process(reply)
    # 直接调用记忆存储
    _state._executor.submit(_auto_remember, user_input, reply)
```

**问题**：修改任一子系统的接口，都需要修改 `_core.py`。无法独立测试任何一个子系统。

#### 问题2：Kotlin 层直接调用 Python 模块

WorldBookActivity、PluginManageActivity 等 Kotlin Activity 通过 Chaquopy 直接调用 `chat_bridge` 模块的具体函数：

```kotlin
// WorldBookActivity.kt — 直接耦合到 chat_bridge 的函数
val py = com.chaquo.python.Python.getInstance()
val module = py.getModule("chat_bridge")
module.callAttr("get_world_book", DEFAULT_BOOK_NAME)
module.callAttr("add_world_book_entry", DEFAULT_BOOK_NAME, entryJson.toString())
module.callAttr("update_world_book_entry", DEFAULT_BOOK_NAME, entry.id, entryJson.toString())
module.callAttr("delete_world_book_entry", DEFAULT_BOOK_NAME, entry.id)
```

**问题**：Kotlin 层和 Python 函数名强耦合。如果 Python 函数改名或签名变化，Kotlin 层编译不会报错，运行时才会崩溃。

#### 问题3：chat_bridge 子模块通过 _state 共享全局状态

```python
# _character.py 使用 _state 中的全局变量
from ._state import _current_character, _CARD_DIR, _ctx

# _core.py 使用 _state 中的全局变量
from ._state import _ctx, _CARD_DIR, _current_params

# _world_book.py 使用 _state 中的全局变量
from ._state import _PYTHON_ROOT
```

**问题**：全局状态使得各模块无法独立实例化，测试时必须模拟整个 `_state` 环境。

#### 问题4：TTS 模块与语音识别耦合在 VoiceController 中

`VoiceController.kt` 同时管理了：
- `SpeechManager`（语音识别 + TTS 合成）
- `VoiceRecorder`（语音录制）
- `VoicePlayer`（语音播放）
- 唤醒词检测

**问题**：TTS 功能无法独立于语音识别使用。如果要替换 TTS 引擎（如从 Sherpa-ONNX 迁移到 VITS），需要修改 `SpeechManager` 和 `VoiceController` 两个类。

#### 问题5：角色卡数据流跨越三层

角色卡数据从 Kotlin 层创建 → JSON 序列化 → Chaquopy 传给 Python → `_character.py` 解析 → 写入 `settings` 和 `_current_character` → 传给 `RolePlayer`。

**问题**：数据流链路长，缺少统一的接口契约。CharacterData (Kotlin) 和 `_current_character` (Python) 字段定义不一致，容易出现字段丢失。

### 1.3 接口不清晰

| 模块 | 当前入口 | 问题 |
|------|---------|------|
| 角色卡 | `chat_bridge._character.set_character_card(json_str)` | 无类型化接口，通过 JSON 字符串传递 |
| 世界书 | `chat_bridge._world_book.list_world_books()` 等 15 个函数 | 平铺导出，无统一入口 |
| 插件 | `chat_bridge._plugins.list_plugins()` 等 4 个函数 | 与 Kotlin 侧 PluginRegistry 概念重叠 |
| TTS | `SpeechManager.startSpeaking()` / `speakSentenceStreaming()` | 与语音识别耦合在 SpeechManager 中 |

### 1.4 测试困难

- **零测试覆盖**：项目中没有任何单元测试
- Python 模块依赖 `src.utils.logger`（需要 Android logcat 环境），无法在纯 Python 环境运行
- Kotlin 模块依赖 Android Context、Chaquopy Python 实例，无法做 JVM 单元测试
- 全局状态（`_state` 单例、`AppContext` 单例）使得测试之间会互相干扰

### 2.2 接口契约设计

#### 2.2.1 通用模块契约 (IModuleContract)

```python
# src/modules/contract.py

from abc import ABC, abstractmethod
from typing import Any

class IModuleContract(ABC):
    """所有功能模块必须实现的接口契约。
    
    设计理念：机器人四肢 — 每个模块是一个可插拔的"肢体"，
    有明确的输入/输出接口，可以独立开发、独立测试。
    """
    
    # ---- 元信息 ----
    @property
    @abstractmethod
    def module_name(self) -> str:
        """模块唯一标识，如 'character', 'world_book', 'plugins', 'tts'"""
        ...
    
    @property
    @abstractmethod
    def module_version(self) -> str:
        """模块版本号"""
        ...
    
    # ---- 生命周期 ----
    @abstractmethod
    def initialize(self, config: dict[str, Any]) -> bool:
        """初始化模块。返回 True 表示成功。"""
        ...
    
    @abstractmethod
    def shutdown(self) -> None:
        """关闭模块，释放资源。"""
        ...
    
    # ---- 状态查询 ----
    @abstractmethod
    def is_ready(self) -> bool:
        """模块是否已初始化并可用。"""
        ...
    
    @abstractmethod
    def health_check(self) -> dict[str, Any]:
        """健康检查，返回 {'status': 'ok', 'details': {...}} 或 {'status': 'error', 'message': '...'}"""
        ...
```

### 2.3 ModuleRegistry 设计

#### 2.3.1 Python 侧 ModuleRegistry

```python
# src/modules/__init__.py

from typing import Any
from src.modules.contract import IModuleContract
from src.utils.logger import get_logger

_log = get_logger()

class ModuleRegistry:
    """模块注册中心。
    
    管理所有功能模块的注册、发现、初始化和生命周期。
    每个模块必须实现 IModuleContract 接口。
    
    使用方式：
        registry = ModuleRegistry()
        registry.register(CharacterModule())
        registry.initialize_all()
        
        # 获取模块
        character = registry.get("character")
        character.set_character({...})
        
        # 关闭所有模块
        registry.shutdown_all()
    """
    
    def __init__(self):
        self._modules: dict[str, IModuleContract] = {}
        self._initialized = False
    
    def register(self, module: IModuleContract) -> bool:
        """注册模块。如果模块名已存在则返回 False。"""
        name = module.module_name
        if name in self._modules:
            _log.warning(f"模块 {name} 已注册，跳过")
            return False
        self._modules[name] = module
        _log.info(f"模块已注册: {name} v{module.module_version}")
        return True
    
    def unregister(self, name: str) -> bool:
        """注销模块。"""
        if name not in self._modules:
            return False
        module = self._modules.pop(name)
        if module.is_ready():
            module.shutdown()
        _log.info(f"模块已注销: {name}")
        return True
    
    def get(self, name: str) -> IModuleContract | None:
        """获取指定模块。"""
        return self._modules.get(name)
    
    def get_all(self) -> dict[str, IModuleContract]:
        """获取所有已注册模块。"""
        return dict(self._modules)
    
    def initialize_all(self, configs: dict[str, dict[str, Any]] | None = None) -> dict[str, bool]:
        """初始化所有已注册模块。
        
        Args:
            configs: 模块配置字典，key 为模块名，value 为配置字典
        
        Returns:
            初始化结果，key 为模块名，value 为是否成功
        """
        configs = configs or {}
        results = {}
        for name, module in self._modules.items():
            config = configs.get(name, {})
            try:
                results[name] = module.initialize(config)
            except Exception as e:
                _log.error(f"模块 {name} 初始化失败: {e}")
                results[name] = False
        self._initialized = True
        return results
    
    def shutdown_all(self) -> None:
        """关闭所有模块，释放资源。"""
        for name, module in reversed(list(self._modules.items())):
            try:
                module.shutdown()
            except Exception as e:
                _log.warning(f"模块 {name} 关闭失败: {e}")
        self._modules.clear()
        self._initialized = False
    
    def health_check_all(self) -> dict[str, dict[str, Any]]:
        """对所有模块执行健康检查。"""
        results = {}
        for name, module in self._modules.items():
            try:
                results[name] = module.health_check()
            except Exception as e:
                results[name] = {"status": "error", "message": str(e)}
        return results


# 全局单例
_registry = ModuleRegistry()

def get_module_registry() -> ModuleRegistry:
    """获取全局模块注册中心实例。"""
    return _registry
```

object ModuleRegistry {
    private val modules = LinkedHashMap<String, Any>()
    
    fun <T> register(name: String, module: T) {
        modules[name] = module as Any
    }
    
    @Suppress("UNCHECKED_CAST")
    fun <T> get(name: String): T? = modules[name] as? T
    
    fun getAll(): Map<String, Any> = LinkedHashMap(modules)

#### 1.2 创建 IModuleContract 接口

文件：`android/app/src/main/python/src/modules/contract.py`

```python
"""模块接口契约 — 所有功能模块必须实现的基础接口"""

from abc import ABC, abstractmethod
from typing import Any


class IModuleContract(ABC):
    """所有功能模块必须实现的接口契约。"""

    # ---- 元信息 ----
    @property
    @abstractmethod
    def module_name(self) -> str:
        """模块唯一标识，如 'character', 'world_book', 'plugins', 'tts'"""
        ...

    @property
    @abstractmethod
    def module_version(self) -> str:
        """模块版本号"""
        ...

    # ---- 生命周期 ----
    @abstractmethod
    def initialize(self, config: dict[str, Any]) -> bool:
        """初始化模块。返回 True 表示成功。"""
        ...

    @abstractmethod
    def shutdown(self) -> None:
        """关闭模块，释放资源。"""
        ...

    # ---- 状态查询 ----
    @abstractmethod
    def is_ready(self) -> bool:
        """模块是否已初始化并可用。"""
        ...

    @abstractmethod
    def health_check(self) -> dict[str, Any]:
        """健康检查。"""
        ...
```

## 5. 风险评估

### 5.1 重构风险矩阵

| 风险 | 影响 | 概率 | 缓解措施 |
|------|:---:|:---:|---------|
| 模块初始化失败导致对话不可用 | 高 | 中 | 每个模块有降级逻辑，初始化失败不影响核心对话 |

| Chaquopy 跨模块调用性能下降 | 中 | 低 | ModuleRegistry 在 Python 侧，无额外跨语言调用 |
| 接口契约变更导致 Kotlin 层编译错误 | 中 | 低 | 保留 `chat_bridge` 函数签名不变 |
| 回归 bug | 高 | 高 | 分步迁移，每步验证后再进行下一步 |
