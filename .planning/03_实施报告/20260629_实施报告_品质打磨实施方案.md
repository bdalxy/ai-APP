**B-02：bm25 score_single 效率优化**（P2，预计 30min）

```python
# bm25.py - score_single 优化：直接计算而非遍历全部
def score_single(self, query: str, doc_id: str) -> float:
    if doc_id not in self._documents:
        return 0.0
    query_tokens = _tokenize_cn(query)
    if not query_tokens:
        return 0.0
    query_tf = Counter(query_tokens)
    score = 0.0
    k1, b, eps = self._config.k1, self._config.b, self._config.epsilon
    doc_len = self._doc_lengths.get(doc_id, 1)
    for term, qt in query_tf.items():
        if term not in self._inverted_index:
            continue
        if doc_id not in self._inverted_index[term]:
            continue
        tf = self._inverted_index[term][doc_id]
        df = self._doc_freq.get(term, 0)
        idf = math.log((self._total_docs - df + 0.5) / (df + 0.5 + eps) + 1.0)
        numerator = tf * (k1 + 1.0)
        denominator = tf + k1 * (1.0 - b + b * doc_len / max(self._avg_doc_length, 1.0))
        score += idf * numerator / denominator * qt
    return score
```

// VoiceController.kt - 使用 AtomicReference 保护回调切换
private val callbackRef = java.util.concurrent.atomic.AtomicReference<SpeechManager.SpeechCallback?>()

private fun startWakeWordRound() {
    // ...
    val originalCallback = callbackRef.get()
    val wakeWordCallback = object : SpeechManager.SpeechCallback {
        // 使用 compareAndSet 确保只恢复一次
        private val restored = java.util.concurrent.atomic.AtomicBoolean(false)
        private fun restoreIfNeeded() {
            if (restored.compareAndSet(false, true)) {
                callbackRef.set(originalCallback)
                speechManager.callback = originalCallback
            }
        }
        override fun onSpeechResult(text: String) {
            restoreIfNeeded()
            // ... 检测唤醒词 ...
        }
        override fun onSpeechError(error: String) {
            restoreIfNeeded()
        }
    }
    callbackRef.set(wakeWordCallback)
    speechManager.callback = wakeWordCallback
    // ...
}
```

fun logNonFatalException(tag: String, throwable: Throwable) {
    try {
        val logDir = getCrashLogDir(context)
        if (!logDir.exists()) logDir.mkdirs()
        val fileName = "nonfatal_${FILE_DATE_FORMAT.format(Date())}.log"
        FileWriter(File(logDir, fileName)).use { writer ->
            writer.write("标签: $tag\n")
            writer.write("时间: ${LOG_DATE_FORMAT.format(Date())}\n")
            writer.write("异常: ${throwable.javaClass.simpleName}: ${throwable.message}\n")
            val sw = StringWriter()
            throwable.printStackTrace(PrintWriter(sw))
            writer.write(sw.toString().lines().take(30).joinToString("\n"))
        }
    } catch (_: Exception) { }
}
```
